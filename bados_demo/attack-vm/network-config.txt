# This file describes the network interfaces available on your system
# and how to activate them. For more information, see interfaces(5).

source /etc/network/interfaces.d/*

# The loopback network interface
auto lo
iface lo inet loopback

# The primary network interface
allow-hotplug enp0s3
iface enp0s3 inet dhcp
# This is an autoconfigured IPv6 interface
iface enp0s3 inet6 auto

# The secondary network interface
allow-hotplug enp0s8
iface enp0s8 inet static

# Good traffic
address 192.168.57.10/24

auto enp0s8:1
allow-hotplug enp0s8:1
iface enp0s8:1 inet static 
address 192.168.57.11/24

# Bad traffic
auto enp0s8:2
allow-hotplug enp0s8:2
iface enp0s8:2 inet static
address 192.168.57.12/24

auto enp0s8:3
allow-hotplug enp0s8:3
iface enp0s8:3 inet static
address 192.168.57.13/24

auto enp0s8:4
allow-hotplug enp0s8:4
iface enp0s8:4 inet static
address 192.168.57.14/24
